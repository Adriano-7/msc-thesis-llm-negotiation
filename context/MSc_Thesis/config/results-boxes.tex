  colback=gray!6,
  colframe=gray!55,
  boxrule=0.5pt,

  left=6pt, right=6pt, top=5pt, bottom=5pt,
  title={\small\bfseries Excerpt~\thetcbcounter: #2},
  coltitle=black,
  colbacktitle=gray!18,
  fonttitle=\sffamily,
  fontupper=\small,

}
