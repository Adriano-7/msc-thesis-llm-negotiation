\section{Draft Diversity and Outcomes} \label{ap6:diversity}

\begin{figure}[ht]
    \centering
    \includegraphics[width=0.7\linewidth]{figures/results/team/draft_diversity_payoff.png}
    \caption{Per-run mean draft diversity (TF-IDF cosine distance, binned) against
    normalised team payoff, by game. Diversity shows no positive association with payoff in
    any game (overall Spearman $\rho \approx -0.05$, n.s.).}
    \label{fig:draft-diversity-payoff}
\end{figure}

\begin{figure}[ht]
    \centering
    \includegraphics[width=0.7\linewidth]{figures/results/team/draft_diversity_winrate.png}
    \caption{Per-run mean draft diversity against team win rate, by game. The association is
    at most weakly negative (overall Spearman $\rho \approx -0.09$, $p = 0.01$).}
    \label{fig:draft-diversity-winrate}
\end{figure}

\begin{figure}[ht]

    of votes, near the $33\%$ chance baseline.}
